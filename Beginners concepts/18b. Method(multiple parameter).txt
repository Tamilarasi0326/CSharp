namespace Methods
{
    class Program
    {
        static void myMethod(string fname, int age)
        {
            Console.WriteLine(fname + " is " + age);
        }

            myMethod("Liam", 5);
            myMethod("Jenny", 8);
            myMethod("Anja", 30);
