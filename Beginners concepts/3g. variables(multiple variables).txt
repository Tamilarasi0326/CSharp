// assign the same value to multiple variables in one line:

using System;

namespace variables
{
    class program3
    {
        static void Main(string[] args)
        {
            int x = 5, y = 6, z = 8;
            Console.WriteLine(x + y + z);
            
            int a, b, c;
            a = b = c = 10;
            Console.WriteLine(a + b + c);
        }
    }
}
