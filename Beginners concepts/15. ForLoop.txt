
using System;

namespace forLoop
{
    class program15
    {
        static void Main(string[] args)
        {
            //print the numbers 0 to 4:
            for(int i = 0; i < 5; i++)
            {
                Console.WriteLine(i);
            }
            
            // print even values between 0 and 10:
            for(int j = 0; j < 11; j = j + 2)
            {
                Console.WriteLine(j);
            }
        }
    }
}
